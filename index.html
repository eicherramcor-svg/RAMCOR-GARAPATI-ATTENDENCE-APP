🔗 Google Sheets Integration
Sync attendance to Google Sheets in real-time. Follow the 3 steps below.
1

Create Google Sheet

Go to sheets.google.com → New Sheet → Name it AP Attendance
Row 1 headers (in this exact order): Timestamp, Organisation, Brand, District, Mandal, Employee ID, Name, Designation, Mobile number, Log in/out, Status, Date, Time, Face Verified, GPS, Map Link

2

Paste Apps Script Code

In the sheet: Extensions → Apps Script
Delete existing code, paste the code below, press Ctrl+S

function doPost(e) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getActiveSheet();
    var d = JSON.parse(e.postData.contents);
    var gps = d.gps || '';
    var mapLink = gps ? ('https://maps.google.com/?q=' + gps) : '';
    sheet.appendRow([
      new Date().toLocaleString('en-IN',{timeZone:'Asia/Kolkata'}), // Timestamp
      d.organisation || '',           // Organisation
      d.brand || '',                  // Brand
      d.district || '',               // District
      d.mandal || '',                 // Mandal
      d.empid || '',                  // Employee ID
      d.name || '',                   // Name
      d.designation || '',            // Designation
      d.mobile || '',                 // Mobile number
      d.status || '',                 // Log in/out  → IN or OUT
      d.subStatus || '',              // Status      → On Duty / Office / Lunch / Permission / Office Closing Time
      d.date || '',                   // Date
      d.time || '',                   // Time
      d.faceVerified ? 'YES' : 'NO',  // Face Verified
      gps,                            // GPS
      mapLink                         // Map Link
    ]);
    return ContentService.createTextOutput(
      JSON.stringify({result:'success'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(
      JSON.stringify({result:'error',message:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
function doGet(e) {
  return ContentService.createTextOutput('AP Attendance OK')
    .setMimeType(ContentService.MimeType.TEXT);
}
